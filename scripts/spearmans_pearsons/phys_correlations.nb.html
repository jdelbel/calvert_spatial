ew_da <- ethel_wind2 %>% 
  mutate(date2 = lubridate::date(date)) %>%

  ungroup() %>% 
  rename(date = date2) %>%
  mutate(w_b1 = rollmeanr(lag(dm_w, 1), k = 3, fill = NA),

  select(date, eth_w = w_b1, eth_wd = wd_b1)
